Expr = Let| Fun | expr
Let  = "Let" + Id + '=' + Expr
Fun  = "Fun" + Id + "->" + Expr + "in" + Expr
expr = term + ('+'|'-' + term)*
term = factor + ('*'|'/' + factor)*
factor = primary | funcall
funcall = primary + (primary)+
primary = Numb | Id | '(' + Expr + ')'
Numb = Digit+
Id = Letter + (Letter | Digit)*
